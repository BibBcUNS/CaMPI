# Archivo de configuración utilizado para la generación de la base de existencias del OpenMP
#___________________________________________________________________________________________
#
# Leer con atención todos los parametros y poner el valor correspondiente.
# IMPORTANTE!: Todos los parametros deben terminar con ;
# El caracter "#" como prefijo de línea indica que la misma es un comentario.

# Indique el nombre de la insitución. Ejemplo:
# Nombre Institucion=Universidad Nacional XYZ;

Nombre Institucion=Institucion XXXXX;

# Indique la ubicación de los archivos correspondientes a la base bibliográfica.
# Debe ser el directorio donde se encuentran, y el nombre de los archivos .mst y .xrf correspondiente. Ejemplo:
# Path base bibliografica=C:\isis\bibun\bibun.*;

Path base bibliografica=C:\biblio3\biblio.*;
	
# Indique el formato de catalogación,
# valores posibles: CEPAL, BIBUN, MARCMP, LILACS, OIT (en minúscula)
# ejemplo: Formato Cataloga=MARC;

Formato Cataloga=MARC;

# Indique el campo y subcampo que contiene el nro. de inventario. Ejemplo:
# Campo Inventario=v852^j;

Campo Inventario=v859^p;

# Indique el campo y subcampo que contiene el valor de sección/colección.
# Este valor será utilizado para definir diferentes políticas de prestamo de acuerdo al material.
# En caso de no utilizar dicho valor especificarlo con vacio.
# En caso de existir dicho valor, el tag (nro de campo) debe ser el mismo que el de inventario.
# Campo Estado=;
# Campo Estado=v852^s;

Campo Estado=v859^b;

# Indique el directorio que contiendrá las base del sistema de circulación
# No incluir la barra al final del path.
# Path bases=c:\apache\www\bases;

Path bases=C:\wamp\www\CaMPI\bases\omp;

# Indique el directorio donde residen los documentos “públicos” accesibles vía HTTP
# El mismo valor configurado en Document_Root dentro de la configuración de apache.
# No incluir la barra al final del path.

Path htdocs=C:\wamp\www\CaMPI\htdocs\omp;

# Indique el directorio en el que se encuentra el utilitario mx y demÃ¡s utilitarios
# Ejemplo:
# Path mx=c:\cisis\mx;

# Path mx=C:\CaMPI\bin\cisis\mx;
Path mx=mx;

# A continuación indicamos si habilitar o no la opción de reserva
# Ejemplos:
# Habilitar reserva=si;
# Habilitar reserva=no;

Habilitar reserva=si;
